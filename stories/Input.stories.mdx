# Input

Inputs let users enter and edit text.

<Canvas>
    <Story name="Default">
        {(args) => {
            const [value, setValue] = useState('');
            const handleValueChange = (event) => {
                const targetValue = event.target.value;
                setValue(targetValue);
                // This function is just for Storybook actions logging
                args.onChange(targetValue);
            };
            return <Input {...args} onChange={handleValueChange} value={value} />;
        }}
    </Story>
</Canvas>
